<!-- Canvas: outermost vertical flex -->
<div class="zc-canvas"
     style="display: flex; flex-direction: column">


  <!-- HEADER (OUTSIDE <form>) -->
  <div class="form-header">
    <span class="form-title">new form 1</span>
    <span class="popupClose"><i>✕</i></span>
  </div>


  <!-- FORM (flex: column, scrollable) -->
  <form id="new_form_1"
        style="display: flex; flex-direction: column;
               flex: 1; overflow-y: auto">


    <!-- ── SECTION 1 ── -->
    <div class="zc-section"
         style="display: flex; flex-direction: column">

      <!-- Section header -->
      <div class="section-group"> ... </div>

      <!-- ROW 1: Name + Single Line -->
      <div class="zc-row"
           style="display: flex; flex-direction: row">

        <div class="zc-cell">
          <div class="form-group">
            <label>Name</label>
            <div class="form-field"> ... </div>
          </div>
        </div>

        <div class="zc-cell">
          <div class="form-group">
            <label>Single Line</label>
            <div class="form-field">
              <input style="width:300px">
            </div>
          </div>
        </div>

      </div><!-- /row 1 -->

    </div><!-- /section 1 -->


    <!-- Section separator -->
    <div class="zc-section-separator"></div>


    <!-- ── SECTION 2 ── -->
    <div class="zc-section">
      <div class="section-group"> ... </div>

      <div class="zc-row">
        <div class="zc-cell"> Multi Line </div>
      </div>

      <div class="zc-row">
        <div class="zc-cell"> Dropdown </div>
        <div class="zc-cell"> Department </div>
      </div>

      <div class="zc-row">
        <div class="zc-cell"> City </div>
        <div class="zc-cell"> State </div>
        <div class="zc-cell"> PIN </div>
      </div>

    </div><!-- /section 2 -->

  </form>


  <!-- FOOTER (OUTSIDE <form>) -->
  <div class="zc-footer-wrapper">
    <button type="button">Cancel</button>
    <input type="submit" form="new_form_1">
  </div>


</div><!-- /zc-canvas -->

      <div class="zc-cell">

          <div class="form-field"> ... </div>

      </div>

        <div class="form-group">
          <label>Single Line</label>
          <div class="form-field">
            <input style="width:300px">
          </div>
        </div>
      </div>

    </div><!-- /row 1 -->


    <!-- ROW 2: Multi Line -->
    <div class="zc-row">
      <div class="zc-cell"> Multi Line </div>
    </div>


    <!-- ROW 3: Dropdown + Department -->
